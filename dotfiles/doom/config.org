(use-package! gptel
  :config
  ;; Claude backend (default)
  (setq gptel-model 'claude-sonnet-4-20250514
        gptel-backend (gptel-make-anthropic "Claude"
                        :stream t
                        :key (lambda ()
                               (string-trim
                                (with-temp-buffer
                                  (insert-file-contents "~/.secrets/anthropic_api_key")
                                  (buffer-string))))))

  ;; Gemini backend
  (gptel-make-gemini "Gemini"
    :stream t
    :key (lambda ()
           (string-trim
            (with-temp-buffer
              (insert-file-contents "~/.secrets/gemini_api_key")
              (buffer-string)))))

  ;; UI
  (setq gptel-default-mode 'org-mode))
