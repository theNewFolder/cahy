;; Dynamically add org-roam Project-tagged notes to agenda
(defun my/org-roam-filter-by-tag (tag-name)
  "Filter org-roam nodes by TAG-NAME."
  (lambda (node)
    (member tag-name (org-roam-node-tags node))))

(defun my/org-roam-list-notes-by-tag (tag-name)
  "Return list of org-roam file paths with TAG-NAME."
  (mapcar #'org-roam-node-file
          (seq-filter
           (my/org-roam-filter-by-tag tag-name)
           (org-roam-node-list))))

(defun my/org-roam-refresh-agenda-list ()
  "Rebuild org-agenda-files from org-roam Project-tagged notes."
  (interactive)
  (setq org-agenda-files
        (delete-dups
         (append
          '("~/org/tasks.org" "~/org/inbox.org" "~/org/habits.org"
            "~/org/projects.org" "~/org/meetings.org")
          (my/org-roam-list-notes-by-tag "Project")))))

;; Build agenda list on startup
(add-hook 'doom-after-init-hook #'my/org-roam-refresh-agenda-list)
